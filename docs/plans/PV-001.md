# PV-001: Pipeline Visualizer Component

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Implement the Pipeline Visualizer interactive component — a 6-step animated walkthrough of Falcon-512 verification with play/pause/step/reset controls, wired into `page.tsx` via `next/dynamic` with `ssr: false`.

**Architecture:** Client-side React component using `@effect-atom/atom-react` hooks to read/write three atoms (`pipelineStepsAtom`, `pipelineActiveStepAtom`, `pipelinePlayingAtom`). State logic: `computeStepStatuses(steps, activeIdx)` maps step index to `pending | active | complete`. A `setInterval` drives auto-play at 2s intervals. The component is dynamically imported with SSR disabled since it depends on browser-only atom registry. A thin wrapper `PipelineSection.tsx` follows the same pattern as `PlaygroundSection.tsx`.

**Tech Stack:** React 19, @effect-atom/atom-react (useAtomValue, useAtomSet, useAtom), Next.js 15 dynamic(), Tailwind CSS v4 with `falcon-*` color tokens

### Step 3: Write minimal implementation of `pipeline-utils.ts`

```typescript
// apps/demo/src/components/interactive/pipeline-utils.ts
import type { PipelineStep } from "@/services/types"

/**
 * Pure function: compute step statuses based on the currently-active step index.
 *
 * - Steps before `activeIdx` → "complete"
 * - Step at `activeIdx` → "active"
 * - Steps after `activeIdx` → "pending"
 * - When `activeIdx` is -1, all steps are "pending"
 * - When `activeIdx` >= steps.length, all steps are "complete"
 */
export function computeStepStatuses(
  steps: readonly PipelineStep[],
  activeIdx: number,
): PipelineStep[] {
  return steps.map((s, i) => ({
    ...s,
    status:
      i < activeIdx
        ? ("complete" as const)
        : i === activeIdx
          ? ("active" as const)
          : ("pending" as const),
  }))
}

/**
 * Sum all step counts in the pipeline.
 */
export function computeTotalSteps(steps: readonly PipelineStep[]): number {
  return steps.reduce((sum, s) => sum + s.stepCount, 0)
}
```

### Step 3: Create initial `PipelineVisualizer.tsx` with `PipelineStepCard`

```tsx
// apps/demo/src/components/interactive/PipelineVisualizer.tsx
"use client"

import { useCallback, useEffect, useRef } from "react"
import { useAtomValue, useAtomSet, useAtom } from "@effect-atom/atom-react"
import {
  pipelineStepsAtom,
  pipelineActiveStepAtom,
  pipelinePlayingAtom,
} from "@/atoms/pipeline"
import type { PipelineStep } from "@/services/types"
import { computeStepStatuses, computeTotalSteps } from "./pipeline-utils"

/** Auto-advance interval in milliseconds */
const PLAY_INTERVAL_MS = 2000

export function PipelineVisualizer() {
  const steps = useAtomValue(pipelineStepsAtom)
  const [activeStep, setActiveStep] = useAtom(pipelineActiveStepAtom)
  const [playing, setPlaying] = useAtom(pipelinePlayingAtom)
  const setSteps = useAtomSet(pipelineStepsAtom)
  const timerRef = useRef<ReturnType<typeof setInterval> | null>(null)

  const updateStepStatuses = useCallback(
    (currentIdx: number) => {
      setSteps((prev) => computeStepStatuses(prev, currentIdx))
    },
    [setSteps],
  )

  // Auto-play timer
  useEffect(() => {
    if (playing) {
      timerRef.current = setInterval(() => {
        setActiveStep((prev) => {
          const next = prev + 1
          if (next >= steps.length) {
            setPlaying(false)
            // Mark all steps complete when finishing
            updateStepStatuses(steps.length)
            return prev
          }
          updateStepStatuses(next)
          return next
        })
      }, PLAY_INTERVAL_MS)
    }
    return () => {
      if (timerRef.current) clearInterval(timerRef.current)
    }
  }, [playing, steps.length, setActiveStep, setPlaying, updateStepStatuses])

  const handlePlay = () => {
    if (activeStep >= steps.length - 1) {
      // Restart from beginning if at end
      setActiveStep(0)
      updateStepStatuses(0)
    } else if (activeStep === -1) {
      // Start from beginning
      setActiveStep(0)
      updateStepStatuses(0)
    }
    setPlaying(true)
  }

  const handlePause = () => {
    setPlaying(false)
  }

  const handleStep = () => {
    if (playing) return
    const next = activeStep + 1
    if (next < steps.length) {
      setActiveStep(next)
      updateStepStatuses(next)
    }
  }

  const handleReset = () => {
    setPlaying(false)
    setActiveStep(-1)
    updateStepStatuses(-1)
  }

  const totalSteps = computeTotalSteps(steps)

  return (
    <section id="pipeline" className="px-6 py-20 lg:px-8">
      <div className="mx-auto max-w-4xl">
        <h2 className="text-3xl font-bold tracking-tight">
          Pipeline Deep-Dive
        </h2>
        <p className="mt-4 text-falcon-muted">
          Step through the on-chain verification algorithm. Total:{" "}
          <span className="font-mono text-falcon-text">
            ~{totalSteps.toLocaleString()}
          </span>{" "}
          Cairo steps.
        </p>

        {/* Controls */}
        <div className="mt-8 flex gap-3">
          {!playing ? (
            <button
              onClick={handlePlay}
              className="rounded-lg bg-falcon-primary px-4 py-2 text-sm font-semibold text-white hover:bg-falcon-primary/80 transition-colors"
              aria-label={activeStep >= 0 ? "Restart pipeline" : "Play pipeline"}
            >
              {activeStep >= 0 ? "Restart" : "Play"}
            </button>
          ) : (
            <button
              onClick={handlePause}
              className="rounded-lg bg-falcon-surface border border-falcon-muted/20 px-4 py-2 text-sm font-semibold hover:bg-falcon-surface/80 transition-colors"
              aria-label="Pause pipeline"
            >
              Pause
            </button>
          )}
          <button
            onClick={handleStep}
            disabled={playing || activeStep >= steps.length - 1}
            className="rounded-lg bg-falcon-surface border border-falcon-muted/20 px-4 py-2 text-sm font-semibold hover:bg-falcon-surface/80 disabled:opacity-50 disabled:cursor-not-allowed transition-colors"
            aria-label="Advance one step"
          >
            Step
          </button>
          <button
            onClick={handleReset}
            className="rounded-lg bg-falcon-surface border border-falcon-muted/20 px-4 py-2 text-sm font-semibold hover:bg-falcon-surface/80 transition-colors"
            aria-label="Reset pipeline"
          >
            Reset
          </button>
        </div>

        {/* Pipeline Steps */}
        <div className="mt-8 space-y-4">
          {steps.map((step, i) => (
            <PipelineStepCard key={step.id} step={step} index={i} />
          ))}
        </div>
      </div>
    </section>
  )
}

function PipelineStepCard({
  step,
  index,
}: {
  readonly step: PipelineStep
  readonly index: number
}) {
  const statusColors: Record<PipelineStep["status"], string> = {
    pending: "border-falcon-muted/20",
    active: "border-falcon-primary ring-2 ring-falcon-primary/20",
    complete: "border-falcon-success/30",
  }

  return (
    <div
      className={`rounded-xl border bg-falcon-surface p-5 transition-all duration-300 ${statusColors[step.status]}`}
      data-testid={`pipeline-step-${step.id}`}
      data-status={step.status}
    >
      <div className="flex items-start justify-between">
        <div className="flex items-center gap-3">
          <span
            className={`flex h-8 w-8 items-center justify-center rounded-full text-sm font-bold ${
              step.status === "complete"
                ? "bg-falcon-success/20 text-falcon-success"
                : step.status === "active"
                  ? "bg-falcon-primary/20 text-falcon-primary"
                  : "bg-falcon-muted/10 text-falcon-muted"
            }`}
          >
            {step.status === "complete" ? "\u2713" : index + 1}
          </span>
          <div>
            <h3 className="font-mono font-semibold">{step.name}</h3>
            <p className="text-sm text-falcon-muted mt-0.5">
              {step.description}
            </p>
          </div>
        </div>
        <span className="text-sm text-falcon-muted font-mono whitespace-nowrap">
          {step.stepCount.toLocaleString()} steps
        </span>
      </div>

      {/* Expanded input/output detail for active step only */}
      {step.status === "active" && (
        <div className="mt-4 grid gap-3 sm:grid-cols-2">
          <div className="rounded-lg bg-falcon-bg p-3">
            <p className="text-xs font-semibold text-falcon-muted uppercase tracking-wider mb-1">
              Input
            </p>
            <p className="text-sm font-mono text-falcon-text/80">
              {step.input}
            </p>
          </div>
          <div className="rounded-lg bg-falcon-bg p-3">
            <p className="text-xs font-semibold text-falcon-muted uppercase tracking-wider mb-1">
              Output
            </p>
            <p className="text-sm font-mono text-falcon-text/80">
              {step.output}
            </p>
          </div>
        </div>
      )}
    </div>
  )
}
```

/**
 * Client-side wrapper for PipelineVisualizer with next/dynamic.
 *
 * `ssr: false` is required because the component uses @effect-atom/atom-react

const PipelineVisualizerDynamic = dynamic(
  () =>
    import("./PipelineVisualizer").then((m) => m.PipelineVisualizer),

    ssr: false,

          <h2 className="text-3xl font-bold tracking-tight">
            Pipeline Deep-Dive
          </h2>
          <div className="mt-8 h-64 rounded-xl bg-falcon-surface animate-pulse" />
        </div>
      </section>
    ),
  },
)

export function PipelineSection() {
  return <PipelineVisualizerDynamic />
}

## Risks and Mitigations

| Risk | Likelihood | Mitigation |
|------|-----------|------------|
| `useAtomSet` callback form `(prev) => next` may not be supported | Low — confirmed from Hooks.d.ts: `(value: W \| ((value: R) => W)) => void` | Confirmed the API accepts updater functions |
| `useAtom` returns `readonly [value, write]` — destructuring may confuse | Low | Use `const [value, setValue] = useAtom(atom)` pattern |
| Timer cleanup race condition (setInterval fires after unmount) | Low | `useEffect` return clears interval; `useRef` ensures stable reference |
| `setActiveStep` inside `setInterval` reads stale closure | Medium | Use functional updater `setActiveStep((prev) => ...)` — atom setter supports this |
| `computeStepStatuses` called with stale `steps` array | Medium — was a bug in the original impl plan (closed over `steps`) | Fixed: use `setSteps((prev) => computeStepStatuses(prev, idx))` functional updater |
| bun:test cannot render React components (no DOM) | Low | Test pure functions only; visual verification for React rendering |
| Existing 24 pipeline atom tests break | Very Low | Tests are isolated via `Registry.make()`; our changes are additive only |
